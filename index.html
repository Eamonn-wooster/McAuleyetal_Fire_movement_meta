1.3 Calculating Effect Sizes

# Calculating Hedge's G
es <- escalc(
  measure = "SMD",       
  m1i = Mean_Fire,
  sd1i = sd_Fire,
  n1i = Burned_n,
  m2i = Mean_Unburned,
  sd2i = sd_Unburned,
  n2i = Unburned_n,
  data = data,
  vtype = "UB")          




# Adding factor for obs level random effect
es$Obs_ID <- factor(1:nrow(es))

# Adding factor for study level random effect
es$StudyID <- NULL

es$Study_ID <- factor(as.numeric(as.factor(es$Firstauthor_Year)))

# Correcting direction of ES for Harris2020 tortuosity
invisible(es %>%
  dplyr::filter(Firstauthor_Year %in% c("Harris2020")))
  
invisible(es <- es %>%
  mutate(yi = ifelse(Obs_ID %in% c(91), yi * -1, yi)))

# Writing function for extracting est, ci.lb, and ci.ub from all models

  df <- data.frame(est = model$b, lower = model$ci.lb, upper = model$ci.ub)
